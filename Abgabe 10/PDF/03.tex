public static bool NaiveSearch(string text, string muster)
{
	int i = 0;
	
	while (text[i] != muster[0] && i < text.Length)
	{
		i++;
	}

	for (int j = 0; j < muster.Length; j++)
	{
    	if (i + j == text.Length - 1 || text[i + j] != muster[j])
    	{
    		return false;
    	}
	}

	return true;
}
\end{minted}
