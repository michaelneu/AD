\section*{Aufgabe 1}
	a) Registermaschinen-Code für $\sum_{i=0}^n i^2$\\
    \begin{minted}[framesep=2mm, linenos, xleftmargin=6mm]{python}
# shorthand python check for correct results (n = 10)
# >>> sum([i**2 for i in range(11)])

LDK 1
STA 100
INP 101

    MUL 101
    ADD 102
    STA 102
    
    LDA 101
    SUB 100
    STA 101

    JGE 8

HLT 99
    \end{minted}
